SELECT 
    c.TABLE_NAME,
    c.ORDINAL_POSITION                                AS poz,
    c.COLUMN_NAME,
    c.DATA_TYPE,
    c.CHARACTER_MAXIMUM_LENGTH                        AS delka,
    c.NUMERIC_PRECISION                               AS presnost,
    c.NUMERIC_SCALE                                   AS meritko,
    c.IS_NULLABLE                                     AS nullable,
    CASE WHEN pk.COLUMN_NAME IS NOT NULL THEN 'PK' ELSE '' END AS klic
FROM INFORMATION_SCHEMA.COLUMNS c
INNER JOIN INFORMATION_SCHEMA.TABLES t
        ON t.TABLE_NAME = c.TABLE_NAME AND t.TABLE_SCHEMA = c.TABLE_SCHEMA
LEFT JOIN (
    SELECT ku.TABLE_NAME, ku.COLUMN_NAME
    FROM INFORMATION_SCHEMA.TABLE_CONSTRAINTS tc
    JOIN INFORMATION_SCHEMA.KEY_COLUMN_USAGE ku
            ON tc.CONSTRAINT_NAME = ku.CONSTRAINT_NAME
           AND tc.TABLE_NAME      = ku.TABLE_NAME
    WHERE tc.CONSTRAINT_TYPE = 'PRIMARY KEY'
) pk ON pk.TABLE_NAME = c.TABLE_NAME AND pk.COLUMN_NAME = c.COLUMN_NAME
WHERE t.TABLE_TYPE = 'BASE TABLE'
ORDER BY c.TABLE_NAME, c.ORDINAL_POSITION;

SELECT 
    s.name        AS schema_name,
    v.name        AS view_name,
    m.definition  AS create_script
FROM sys.views v
JOIN sys.schemas s ON s.schema_id = v.schema_id
JOIN sys.sql_modules m ON m.object_id = v.object_id
ORDER BY s.name, v.name;

dbo	v_961_vysledek_vyvoj	CREATE VIEW dbo.v_961_vysledek_vyvoj
AS
WITH agg AS (SELECT        YEAR(DatZdPln) AS rok, SUM(CASE WHEN LEFT(SynAu, 1) = '5' THEN KcUD ELSE 0 END) AS naklady, SUM(CASE WHEN LEFT(SynAu, 1) = '6' THEN KcUD ELSE 0 END) AS vynosy, 
                                                        SUM(CASE WHEN SynAu = '601001' THEN KcUD ELSE 0 END) AS trzby_bio, SUM(CASE WHEN SynAu = '601002' THEN KcUD ELSE 0 END) AS trzby_nebio, SUM(CASE WHEN SynAu BETWEEN '601000' AND 
                                                        '601099' THEN KcUD ELSE 0 END) AS trzby_celkem
                               FROM            dbo.v_951_pud_vysledovka_zdroj
                               GROUP BY YEAR(DatZdPln))
    SELECT        TOP (100) PERCENT DATETIMEFROMPARTS(rok, 1, 1, 0, 0, 0, 0) AS dt, rok, naklady, vynosy, trzby_bio, trzby_nebio, trzby_celkem, CAST(trzby_bio AS decimal(18, 2)) / NULLIF (CAST(trzby_celkem AS decimal(18, 2)), 0) 
                              AS trzby_bio_podil
     FROM            agg AS agg_1
     WHERE        (rok IS NOT NULL)
     ORDER BY dt
